// Notificador de novas atividades
function checkNewActivities() {
    const activities = document.querySelectorAll('.new-activity');
    if(activities.length > 0) {
        alert(`Você tem ${activities.length} novas atividades!`);
        // Envia notificação para o Discord
        fetch('WEBHOOK_URL', {
            method: 'POST',
            body: JSON.stringify({
                content: `Novas atividades disponíveis!`
            })
        });
    }
}
setInterval(checkNewActivities, 60000);
